  5 
  6 BREAKOUT.namespace('BREAKOUT.filters.Scaler');
  7 
  8 BREAKOUT.filters.Scaler = (function() {
  9 	"use strict";
 10 
 11 	var Scaler;
 12 
 13 	// dependencies
 14 	var FilterBase = BREAKOUT.filters.FilterBase;
 15 
 16 	/**
 17 	 * Scales up an input value from its min and max range to a specified minimum to maximum range. 
 18 	 * A number of scaling functions are provided.	
 19 	 *
 20 	 * @exports Scaler as BREAKOUT.filters.Scaler
 21 	 * @constructor
 22 	 * @augments BREAKOUT.filters.FilterBase
 23 	 * @param {Number} inMin minimum input value
 24 	 * @param {Number} inMax maximum input value
 25 	 * @param {Number} outMin minimum output value
 26 	 * @param {Number} outMax maximum output value
 27 	 * @param {Function} type The function used to map the input curve
 28 	 * @param {Boolean} limiter Whether or not to restrict the input value if it
 29 	 * exceeds the specified range.
 30 	 */
 31 	Scaler = function(inMin, inMax, outMin, outMax, type, limiter) {
 32 
 33 		this.name = "Scaler";
 34 
 35 		this._inMin = inMin || 0;
 36 		this._inMax = inMax || 1;
 37 		this._outMin = outMin || 0;
 38 		this._outMax = outMax || 1;
 39 		this._type = type || this.LINEAR;
 40 		this._limiter = limiter || true;
 41 
 42 	};
 43 
 44 
 45 	Scaler.prototype = BREAKOUT.inherit(FilterBase.prototype);
 46 	Scaler.prototype.constructor = Scaler;
 47 
 48 	/**
 49 	 * Override FilterBase.processSample
 50 	 *
 51 	 * @inheritDoc
 52 	 */
 53 	Scaler.prototype.processSample = function(val) {
 54 		var inRange = this._inMax - this._inMin;
 55 		var outRange = this._outMax - this._outMin;
 56 		var normalVal = (val - this._inMin) / inRange;
 57 		if (this._limiter) {
 58 			normalVal = Math.max(0, Math.min(1, normalVal));
 59 		}
 60 
 61 		return outRange * this._type(normalVal) + this._outMin;
 62 	};
 63 
 64 	/**
 65 	 * y = x
 66 	 */
 67 	Scaler.prototype.LINEAR = function(val) {
 68 		return val;
 69 	};
 70 
 71 	/**
 72 	 * y = x * x
 73 	 */
 74 	Scaler.prototype.SQUARE = function(val) {
 75 		return val;
 76 	};
 77 
 78 	/**
 79 	 * y = sqrt(x)
 80 	 */
 81 	Scaler.prototype.SQUARE_ROOT = function(val) {
 82 		return val;
 83 	};
 84 	
 85 	/**
 86 	 * y = x^4
 87 	 */
 88 	Scaler.prototype.CUBE = function(val) {
 89 		return val;
 90 	};
 91 	
 92 	/**
 93 	 * y = pow(x, 1/4)
 94 	 */
 95 	Scaler.prototype.CUBE_ROOT = function(val) {
 96 		return val;
 97 	};			
 98 
 99 
100 	return Scaler;
101 
102 }());
