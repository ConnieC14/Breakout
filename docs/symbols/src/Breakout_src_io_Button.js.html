 44 BREAKOUT.namespace('BREAKOUT.io.Button');
 45 
 46 BREAKOUT.io.Button = (function() {
 47 
 48 	var Button;
 49 
 50 	// dependencies
 51 	var PhysicalInputBase = BREAKOUT.PhysicalInputBase,
 52 		Event = BREAKOUT.Event,
 53 		Pin = BREAKOUT.Pin,
 54 		ButtonEvent = BREAKOUT.io.ButtonEvent;
 55 
 56 	/**
 57 	 * An object to represent a physical button. The advantage of using the Button class over
 58 	 * listening for pin change events on a Pin object, is that the Button class handles debouncing 
 59 	 * and provides helpful button events: Pressed, Released, Long Press and Sustained Press
 60 	 *
 61 	 * <p>PULL_UP vs PULL_DOWN. If the other end of the resistor connected to the button is 
 62 	 * connected to ground, configuration is PULL_DOWN, if the resistor is connected to power, 
 63 	 * then the configuration is PULL_UP.</p>
 64 	 *
 65 	 * @exports Button as BREAKOUT.io.Button
 66 	 * @constructor
 67 	 * @augments BREAKOUT.PhysicalInputBase
 68 	 * @param {IOBoard} board A reference to the IOBoard instance
 69 	 * @param {Pin} pin A reference to the Pin the button is connected to.
 70 	 * @param {Number} buttonMode The mode of the button (either Button.PULL_DOWN or 
 71 	 * Button.PULL_UP if wired with external resistors or Button.INTERNAL_PULL_UP if
 72 	 * using the internal pull-up resistors. Default is PULL_DOWN.
 73 	 * @param {Number} sustainedPressInterval The delay time in milliseconds before a sustained press event is fired.
 74 	 *
 75 	 */
 76 	Button = function(board, pin, buttonMode, sustainedPressInterval) {
 77 		"use strict";
 78 		
 79 		PhysicalInputBase.call(this);
 80 
 81 		this.name = "Button";
 82 		this._pin = pin;
 83 
 84 		var pinNumber = pin.number;
 85 		
 86 		this.buttonMode = buttonMode || Button.PULL_DOWN;
 87 		this._sustainedPressInterval = sustainedPressInterval || 1000;
 88 
 89 		this._debounceInterval = 20,
 90 		this._repeatCount = 0,
 91 		this._timer = null,
 92 		this._timeout = null;
 93 		
 94 		this._board = board;
 95 		board.setDigitalPinMode(pinNumber, Pin.DIN);
 96 
 97 		if (this.buttonMode === Button.INTERNAL_PULL_UP) {
 98 			// enable internal pull up resistor
 99 			board.enablePullUp(pinNumber);
100 			// set value to high to avoid initial change event
101 			this._pin.value = Pin.HIGH;
102 		}
103 				
104 		this._pin.addEventListener(Event.CHANGE, this.onPinChange.bind(this));	
105 	};
106 
107 
108 	Button.prototype = BREAKOUT.inherit(PhysicalInputBase.prototype);
109 	Button.prototype.constructor = Button;
110 
111 	/**
112 	 * @private
113 	 */
114 	Button.prototype.onPinChange = function(evt) {
115 		
116 		var btnVal = evt.target.value;
117 		var stateHandler;
118 				
119 		if (this.buttonMode === Button.PULL_DOWN) {
120 			if (btnVal === 1) stateHandler = this.pressed;
121 			else stateHandler = this.released;
122 		} else if (this.buttonMode === Button.PULL_UP || this.buttonMode === Button.INTERNAL_PULL_UP) {
123 			if (btnVal === 1) stateHandler = this.released;
124 			else stateHandler = this.pressed;
125 		}
126 		
127 		if (this._timeout === null) {
128 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
129 		} else {
130 			clearTimeout(this._timeout);
131 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
132 		}
133 	};
134 	
135 	/**
136 	 * @private
137 	 */
138 	Button.prototype.pressed = function() {
139 		this._timeout = null;
140 
141 		this.dispatchEvent(new ButtonEvent(ButtonEvent.PRESS));
142 		
143 		this._timer = setInterval(this.sustainedPress.bind(this), this._sustainedPressInterval);
144 	};
145 	
146 	/**
147 	 * @private
148 	 */	
149 	Button.prototype.released = function() {
150 		this._timeout = null;
151 		this.dispatchEvent(new ButtonEvent(ButtonEvent.RELEASE));
152 		
153 		if (this._timer != null) {
154 			clearInterval(this._timer);
155 			this._timer = null;
156 		}
157 		
158 		this._repeatCount = 0;
159 	};
160 	
161 	/**
162 	 * @private
163 	 */
164 	Button.prototype.sustainedPress = function() {
165 		if (this._repeatCount > 0) {
166 			this.dispatchEvent(new ButtonEvent(ButtonEvent.SUSTAINED_PRESS));
167 		} else {
168 			this.dispatchEvent(new ButtonEvent(ButtonEvent.LONG_PRESS));
169 		}
170 		
171 		this._repeatCount++;
172 	};
173 	
174 	/**
175 	 * The debounce time interval in milliseconds.
176 	 * @name Button#debounceInterval
177 	 * @property
178 	 * @type Number
179 	 */ 
180 	Button.prototype.__defineGetter__("debounceInterval", function() { return this._debounceInterval; });
181 	Button.prototype.__defineSetter__("debounceInterval", function(interval) { this._debounceInterval = interval; });
182 	
183 	/**
184 	 * The delay time (in milliseconds) the button must be held before a sustained press event is fired.
185 	 * @name Button#sustainedPressInterval
186 	 * @property
187 	 * @type Number
188 	 */ 
189 	Button.prototype.__defineGetter__("sustainedPressInterval", function() { return this._sustainedPressInterval; });
190 	Button.prototype.__defineSetter__("sustainedPressInterval", function(intervalTime) { this._sustainedPressInterval = intervalTime; });
191 
192 	/**
193 	 * [read-only] The pin number of the pin the button is attached to.
194 	 * @name Button#pinNumber
195 	 * @property
196 	 * @type Number
197 	 */
198 	Button.prototype.__defineGetter__("pinNumber", function() { return this._pin.number; });	
199 
200 	/** @constant */
201 	Button.PULL_DOWN = 0;
202 	/** @constant */
203 	Button.PULL_UP 	= 1;
204 	/** @contstant */
205 	Button.INTERNAL_PULL_UP = 2;
206 
207 	return Button;
208 
209 }());
