  8 BO.WSocketEvent = (function() {
  9 	
 10 	var WSocketEvent;
 11 
 12 	// dependencies
 13 	var Event = JSUTILS.Event;
 14 
 15 	/**
 16 	 * @exports WSocketEvent as BO.WSocketEvent
 17 	 * @constructor
 18 	 * @augments JSUTILS.Event
 19 	 * @param {String} type The event type
 20 	 */
 21 	WSocketEvent = function(type) {
 22 		this.name = "WSocketEvent"; // for testing
 23 		
 24 		// call the super class
 25 		// 2nd parameter is passed to EventDispatcher constructor
 26 		Event.call(this, type);		
 27 	};
 28 
 29 	// events
 30 	/** @constant */
 31 	WSocketEvent.CONNECTED = "webSocketConnected";
 32 	/** @constant */
 33 	WSocketEvent.MESSAGE = "webSocketMessage";
 34 	/** @constant */
 35 	WSocketEvent.CLOSE = "webSocketClosed";
 36 
 37 	WSocketEvent.prototype = JSUTILS.inherit(Event.prototype);
 38 	WSocketEvent.prototype.constructor = WSocketEvent;	
 39 
 40 	return WSocketEvent;

 45 /**
 46  * Wrapper for various client-side Websocket implementation to 
 47  * provide a consistent interface
 48  *
 49  * @author jeff hoefs
 50  */
 51 
 52 JSUTILS.namespace('BO.WSocketWrapper');
 53 
 54 BO.WSocketWrapper = (function() {
 55 	"use strict";
 56 
 57  	var WSocketWrapper;
 58 
 59  	// dependencies
 60  	var EventDispatcher = JSUTILS.EventDispatcher,
 61  		WSocketEvent = BO.WSocketEvent;
 62 
 63 	/**
 64 	 * Creates a wrapper for various websocket implementations to unify the interface.
 65 	 *
 66 	 * @exports WSocketWrapper as BO.WSocketWrapper
 67 	 * @constructor
 68 	 * @param {String} host The host address of the web server.
 69 	 * @param {Number} port The port to connect to on the web server.
 70 	 * @param {Boolean} useSocketIO Set true to use socket.io implementation, set false to use
 71 	 * native websocket implementation.
 72 	 * @param {String} protocol The websockt protocol definition (if necessary).
 73 	 */
 74 	WSocketWrapper = function(host, port, useSocketIO, protocol) {
 75 		this.name = "WSocketWrapper"; // for testing
 76 
 77 		EventDispatcher.call(this, this);
 78 
 79 		this._host = host;
 80 		this._port = port;
 81 		this._protocol = protocol || "default-protocol";
 82 		this._useSocketIO = useSocketIO || false;
 83 		this._socket = null;
 84 		this._readyState = ""; // only applies to native WebSocket implementations
 85 
 86 		this.init(this);
 87 
 88 	};
 89 
 90 	WSocketWrapper.prototype = JSUTILS.inherit(EventDispatcher.prototype);
 91 	WSocketWrapper.prototype.constructor = WSocketWrapper;
 92 
 93 	/**
 94 	 * Initialize the websocket
 95 	 * @param {Object} self A reference to this websocket object.
 96 	 * @private
 97 	 */
 98 	WSocketWrapper.prototype.init = function(self) {
 99 
100 		if (self._useSocketIO) {
101 			self._socket = io.connect("http://"+self._host+":"+self._port);
102 
103 			try {
104 				/** @private */
105 				self._socket.on('connect', function() {
106 					self.dispatchEvent(new WSocketEvent(WSocketEvent.CONNECTED));
107 					/** @private */
108 					self._socket.on('message', function(msg) {
109 						self.dispatchEvent(new WSocketEvent(WSocketEvent.MESSAGE), {message: msg});
110 					});
111 				});
112 				
113 			} catch(exception){
114 				console.log("Error "+exception);
115 			}
116 
117 		} else {
118 			
119 			try {
120 
121 				if ("MozWebSocket" in window) {
122 					self._socket = new MozWebSocket("ws://"+self._host+":"+self._port, self._protocol);
123 				} else if ("WebSocket" in window) {
124 					// Safari doesn't like protocol parameter
125 					//self._socket = new WebSocket("ws://"+self._host+":"+self._port, self._protocol);
126 					self._socket = new WebSocket("ws://"+self._host+":"+self._port);
127 				} else {
128 					console.log("Websockets not supported by this browser");
129 					throw "Websockets not supported by this browser";
130 				}
131 				/** @private */
132 				self._socket.onopen = function() {
133 
134 					self.dispatchEvent(new WSocketEvent(WSocketEvent.CONNECTED));
135 					/** @private */
136 					self._socket.onmessage = function(msg) {
137 						self.dispatchEvent(new WSocketEvent(WSocketEvent.MESSAGE), {message: msg.data});
138 					};
139 					/** @private */
140 					self._socket.onclose = function() {
141 						self._readyState = self._socket.readyState;
142 						self.dispatchEvent(new WSocketEvent(WSocketEvent.CLOSE));	
143 					};
144 
145 				};
146 				
147 			} catch(exception){
148 				console.log("Error "+exception);
149 			}
150 				
151 		}
152 
153 	};
154 
155 	/**
156 	 * Send a message
157 	 * @param {String} message The message to send
158 	 */
159 	WSocketWrapper.prototype.send = function(message) {
160 		// to do: ensure socket is not null before trying to send
161 		this._socket.send(message);
162 	};
163 
164 	// to do: ensure socket is not null before trying to get readyState
165 
166 	/**
167 	 * [read-only] Wrapper for the readyState method of the native websocket implementation
168 	 * <p>CONNECTING = 0, OPEN = 1, CLOSING = 2, CLOSED = 3</p>
169 	 * @name WSocketWrapper#readyState
170 	 * @property
171 	 * @type String
172 	 */		 
173 	WSocketWrapper.prototype.__defineGetter__("readyState", function() { return this._readyState; });
174 
175 
176 	// document events
177 
178 	/**
179 	 * The webSocketConnected event is dispatched when a connection with
180 	 * the websocket is established.
181 	 * @name WSocketWrapper#webSocketConnected
182 	 * @type BO.WebsocketEvent.CONNECTED
183 	 * @event
184 	 * @param {BO.WSocketWrapper} target A reference to the WSocketWrapper object.
185 	 */	
186 
187 	/**
188 	 * The webSocketMessage event is dispatched when a websocket message is received.
189 	 * @name WSocketWrapper#webSocketMessage
190 	 * @type BO.WebsocketEvent.MESSAGE
191 	 * @event
192 	 * @param {BO.WSocketWrapper} target A reference to the WSocketWrapper object.
193 	 * @param {String} message The websocket data	 
194 	 */	
195 
196 	/**
197 	 * The webSocketClosed event is dispatched the websocket connection is closed.
198 	 * @name WSocketWrapper#webSocketClosed
199 	 * @type BO.WebsocketEvent.CLOSE
200 	 * @event
201 	 * @param {BO.WSocketWrapper} target A reference to the WSocketWrapper object. 
202 	 */		 
203 
204 	return WSocketWrapper;
205 
206 }());
