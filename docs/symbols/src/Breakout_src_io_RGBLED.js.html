  5 
  6 BREAKOUT.namespace('BREAKOUT.io.RGBLED');
  7 
  8 BREAKOUT.io.RGBLED = (function() {
  9 
 10 	var RGBLED;
 11 
 12 	// dependencies
 13 	var Pin = BREAKOUT.Pin,
 14 		LED = BREAKOUT.io.LED;
 15 
 16 	/**
 17 	 * The RGB pins of the RGBLED must be connected to PWM pins on the IOBoard.
 18 	 *
 19 	 * <p>COMMON_ANODE vs COMMON_CATHODE. You can determine if your RGB LED is 
 20 	 * common anode or common cathode by reading the datasheet. To wire a 
 21 	 * common cathode RGBLED, connect the cathode to ground and the 3 anode pins 
 22 	 * to the IOBoard PWM pins via 330 ohm resistors. For a common anode LED, the 
 23 	 * anode is connected to power and the 3 cathode pins are connected to the 
 24 	 * IOBoard PWM pins via 330 ohm resistors.</p>
 25 	 *
 26 	 * @exports RGBLED as BREAKOUT.io.RGBLED
 27 	 * @constructor
 28 	 * @param {IOBoard} board A reference to the IOBoard instance that the servo is attached to.
 29 	 * @param {Pin} redLEDPin A reference to the IOBoard Pin the red led pin is connected to.
 30 	 * @param {Pin} greenLEDPin A reference to the IOBoard Pin the green led pin is connected to.
 31 	 * @param {Pin} blueLEDPin A reference to the IOBoard Pin the blue led pin is connected to.	 
 32 	 * @param {Number} driveMode The drive mode of the RGBLED. Must be set to RGBLED.COMMON_ANODE
 33 	 * or RGBLED.COMMON_CATHODE. RGBLED.COMMON_ANODE is default.
 34 	 */
 35 	RGBLED = function(board, redLEDPin, greenLEDPin, blueLEDPin, driveMode) {
 36 		"use strict";
 37 		
 38 		this.name = "RGBLED";
 39 
 40 		if (driveMode === undefined) driveMode = RGBLED.COMMON_ANODE;
 41 
 42 		this._redLED = new LED(board, redLEDPin, driveMode);
 43 		this._greenLED = new LED(board, greenLEDPin, driveMode);
 44 		this._blueLED = new LED(board, blueLEDPin, driveMode);
 45 	};
 46 
 47 	RGBLED.prototype = {
 48 
 49 		/**
 50 		 * Set the RGBLED color
 51 		 * @param {Number} red The red value (0 - 255)
 52 		 * @param {Number} green The green value (0 - 255)
 53 		 * @param {Number} blue The blue value (0 - 255)
 54 		 */
 55 		setColor: function(red, green, blue) {
 56 			red = red / 255;
 57 			green = green / 255;
 58 			blue = blue / 255;
 59 
 60 			this._redLED.intensity = red;
 61 			this._greenLED.intensity = green;
 62 			this._blueLED.intensity = blue;
 63 		},
 64 
 65 		/**
 66 		 * Fade in the RGBLED from the off state
 67 		 * @param {Number} time The time of the fade (in milliseconds)
 68 		 */
 69 		fadeIn: function(time) {
 70 			time = time || 1000;
 71 			this._redLED.fadeTo(1, time);
 72 			this._greenLED.fadeTo(1, time);
 73 			this._blueLED.fadeTo(1, time);
 74 		},
 75 
 76 		/**
 77 		 * Fade out the RGBLED from the on state
 78 		 * @param {Number} time The time of the fade (in milliseconds)
 79 		 */
 80 		fadeOut: function(time) {
 81 			time = time || 1000;
 82 			this._redLED.fadeTo(0, time);
 83 			this._greenLED.fadeTo(0, time);
 84 			this._blueLED.fadeTo(0, time);			
 85 		},
 86 
 87 		/**
 88 		 * Fade from the current color to the new color
 89 		 * @param {Number} red The red value to fade to (0 - 255)
 90 		 * @param {Number} green The green value to fade to (0 - 255)
 91 		 * @param {Number} blue The blue value to fade to (0 - 255)
 92 		 * @param {Number} time The time of the fade (in milliseconds)		 
 93 		 */		
 94 		fadeTo: function(red, green, blue, time) {
 95 			red = red / 255;
 96 			green = green / 255;
 97 			blue = blue / 255;
 98 			time = time || 1000;
 99 
100 			this._redLED.fadeTo(red, time);
101 			this._greenLED.fadeTo(green, time);
102 			this._blueLED.fadeTo(blue, time);
103 		}
104 			
105 	};
106 
107 	/** @constant */
108 	RGBLED.COMMON_ANODE = LED.SYNC_DRIVE;
109 	/** @constant */
110 	RGBLED.COMMON_CATHODE = LED.SOURCE_DRIVE;				
111 
112 	return RGBLED;
113 
114 }());
