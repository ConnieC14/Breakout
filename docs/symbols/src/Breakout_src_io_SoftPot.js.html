 10 	var SoftPotEvent;

 12 	// dependencies
 13 	var Event = JSUTILS.Event;
 14 
 15 	/**
 16 	 * @exports SoftPotEvent as BO.io.SoftPotEvent
 17 	 * @constructor
 18 	 * @augments JSUTILS.Event
 19  	 * @param {String} type The event type
 20  	 * @param {Number} touchPoint The value where the softpot was touched	 
 21 	 */
 22 	SoftPotEvent = function(type, touchPoint) {
 23 
 24 		this.name = "SoftPotEvent";
 25 
 26 		Event.call(this, type);
 27 		this._touchPoint = touchPoint;
 28 	};

 44 	SoftPotEvent.prototype.constructor = SoftPotEvent;

 48 	 * @name SoftPotEvent#value

 65 JSUTILS.namespace('BO.io.SoftPot');

 67 BO.io.SoftPot = (function() {
 68 
 69 	var SoftPot;
 70 
 71 	// private static constants:
 72 	var	TAP_TIMEOUT				= 200,
 73 		FLICK_TIMEOUT			= 200,
 74 		PRESS_TIMER_INTERVAL	= 10,
 75 		DEBOUNCE_TIMEOUT		= 20;	
 76 
 77 	// dependencies
 78 	var PhysicalInputBase = BO.PhysicalInputBase,
 79 		Pin = BO.Pin,
 80 		Event = JSUTILS.Event,
 81 		Scaler = BO.filters.Scaler,
 82 		Timer = JSUTILS.Timer,
 83 		TimerEvent = JSUTILS.TimerEvent,
 84 		SoftPotEvent = BO.io.SoftPotEvent;
 85 
 86 	/**
 87 	 * A softpot analog sensor.
 88 	 *
 89 	 * @exports SoftPot as BO.io.SoftPot
 90 	 * @constructor
 91 	 * @augments BO.PhysicalInputBase
 92 	 * @param {IOBoard} board A reference to the IOBoard instance
 93 	 * @param {Pin} pin A reference to the Pin the softpot is connected to.
 94 	 * @param {Number} softPotLength The length of the softpot in mm 
 95 	 *
 96 	 */
 97 	SoftPot = function(board, pin, softPotLength) {
 98 		"use strict";
 99 		
100 		PhysicalInputBase.call(this);
101 
102 		this.name = "SoftPot";
103 		this._pin = pin;
104 
105 		softPotLength = softPotLength || 100;
106 
107 		this._isDrag = false;
108 		this._flickDistance = 0;
109 		this._touchPoint = 0;
110 		this._lastMovePoint = 0;
111 		this._isTouched = false;
112 		this._distanceFromPressed = 0;
113 		this._minFlickMovement = 1.0/softPotLength * 2.5;
114 		this._minDragMovement = 1.0/softPotLength * 1.0;
115 		this._tapTimeout = TAP_TIMEOUT;
116 		
117 		this._board = board;
118 		board.enableAnalogPin(this._pin.analogNumber);
119 
120 		this._debugMode = false;
121 						
122 		this._pin.addEventListener(Event.CHANGE, this.onPinChange.bind(this));
123 
124 		this._pressTimer = new Timer(PRESS_TIMER_INTERVAL, 0);
125 		this._flickTimer = new Timer(FLICK_TIMEOUT, 1);
126 	};
127 
128 
129 	SoftPot.prototype = JSUTILS.inherit(PhysicalInputBase.prototype);
130 	SoftPot.prototype.constructor = SoftPot;
131 
132 	/**
133 	 * @private
134 	 * @param {Event} evt Event.CHANGE
135 	 */
136 	SoftPot.prototype.onPinChange = function(evt) {
137 		var val = evt.target.value;
138 
139 		if (val === 0) {
140 			this.onRelease();
141 		} else {
142 			if (!this._isTouched) {
143 				this.startTouch(val);
144 				this._lastMovePoint = val;
145 			} else {
146 				this.onMove(val);
147 			}
148 		}
149 	};
150 
151 	/**
152 	 * @private
153 	 * @param {Number} touchPoint The value where the touch is occuring on the strip
154 	 */
155 	SoftPot.prototype.setMinFlickMovement = function(num) {
156 		this._minFlickMovement = num;	
157 	};
158 	
159 	/**
160 	 * @private
161 	 */
162 	SoftPot.prototype.startTouch = function(touchPoint) {
163 		
164 		this._pressTimer.reset();
165 		this._pressTimer.start();
166 		
167 		// where we pressed
168 		this._touchPoint = touchPoint;
169 		this.dispatch(SoftPotEvent.PRESS);
170 		
171 		this._isTouched = true;
172 		this._isDrag = false;	
173 	};
174 
175 	/**
176 	 * @private
177 	 */
178 	SoftPot.prototype.onRelease = function() {		
179 
180 			var dispatchedFlick = false; 
181 			
182 			// discard unintentional touch / noise
183 			if (this._pressTimer.currentCount > DEBOUNCE_TIMEOUT / PRESS_TIMER_INTERVAL) {
184 				// must meet minimum time requirement for flick  
185 				if (this._flickTimer.running) {
186 					if (this._flickDir > 0) {
187 						this.dispatch(SoftPotEvent.FLICK_DOWN);
188 					} else {
189 						this.dispatch(SoftPotEvent.FLICK_UP);
190 					}	
191 					dispatchedFlick = true; 
192 					
193 				} 
194 							
195 				if (!dispatchedFlick) {
196 			
197 					// check for presses  
198 					if (this._pressTimer.running) {
199 				
200 						// if less than tap timeout, then it is a tap 
201 						if (!this._isDrag && this._pressTimer.currentCount <= this._tapTimeout / PRESS_TIMER_INTERVAL) {
202 							this.dispatch(SoftPotEvent.TAP);
203 						}
204 					} 
205 				}
206 			}
207 
208 			this.dispatch(SoftPotEvent.RELEASE);
209 
210 			this.resetForNext(); 
211 	};
212 	
213 	/**
214 	 * @private
215 	 * @param {Number} touchPoint The value where the touch is occuring on the strip
216 	 */
217 	SoftPot.prototype.onMove = function(touchPoint) {		
218 	
219 		this._touchPoint = touchPoint;		
220 					
221 		// save current point 		
222 		var curMovePoint = touchPoint; 
223 		
224 		// flick handeling 			
225 		this._flickDistance = Math.abs(curMovePoint - this._lastMovePoint) 
226 		
227 		if (!this._isDrag && this._flickDistance > this._minFlickMovement ) {
228 			this._flickTimer.reset(); 
229 			this._flickTimer.start(); 
230 			
231 			if(curMovePoint - this._lastMovePoint > 0) {
232 				this._flickDir = 1;
233 			} else {
234 				this._flickDir = -1;
235 			}
236 			
237 			this._isDrag = false; 
238 		}			
239 		
240 		var dragDistance = Math.abs(curMovePoint - this._lastMovePoint);				
241 
242 		// dragging handler 
243 		// dont check when flick timer is running
244 		//console.log("min drag = " + this._minDragMovement);
245 		if ((dragDistance > this._minDragMovement) && (this._flickTimer.running === false)) {
246 			this._isDrag = true; 
247 		}
248 		
249 		if (this._isDrag) {
250 			this.dispatch(SoftPotEvent.DRAG);
251 			this._distanceFromPressed = curMovePoint - this._lastMovePoint;
252 		}
253 								
254 		this.debug("SoftPot: distance traveled flick is " + this._flickDistance); 
255 		this.debug("SoftPot: distance traveled drag is " + dragDistance); 
256 
257 		
258 		// reuse for next 
259 		this._lastMovePoint = curMovePoint; 
260 			
261 	};
262 
263 	/**
264 	 * Scale from the minimum and maximum input values to 0.0 -> 1.0.
265 	 *
266 	 * @param {Number} minimum The minimum value
267 	 * @param {Number} maximum The maximum value
268 	 */
269 	SoftPot.prototype.setRange = function(minimum, maximum) {
270 		this._pin.removeAllFilters();
271 		this._pin.addFilter(new Scaler(mimimum, maximum, 0, 1, Scaler.LINEAR));	
272 	};
273 
274 	/**
275 	 * @private
276 	 * @type {Event} type The event type
277 	 */
278 	SoftPot.prototype.dispatch = function(type) {
279 		this.debug("SoftPot dispatch " + type);
280 		this.dispatchEvent(new SoftPotEvent(type, this._touchPoint));	
281 	};
282 
283 	/**
284 	 * reset whenever you need the next Touch point
285 	 * @private
286 	 */
287 	SoftPot.prototype.resetForNext = function() {
288 		this._flickTimer.stop();
289 		this._pressTimer.stop();
290 		this._isTouched = false;
291 		this._isDrag = false;
292 	};
293

295 	 * for debugging
296 	 * @private
297 	 */
298 	SoftPot.prototype.debug = function(str) {
299 		if (this._debugMode) {
300 			console.log(str); 
301 		}
302 	};
