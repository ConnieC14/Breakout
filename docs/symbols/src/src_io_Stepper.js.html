  6 JSUTILS.namespace('BO.io.Stepper');
  7 
  8 BO.io.Stepper = (function() {
  9 	"use strict";
 10 
 11 	var Stepper;
 12 	var instanceCounter = 0;
 13 
 14 	// private static constants
 15 	var CONFIG = 0,
 16 		STEP = 1,
 17 		SPEED = 2,
 18 		MAX_STEPS = 16383,
 19 		MAX_SPEED = 16383;
 20 
 21 	// dependencies
 22 	var Pin = BO.Pin,
 23 		EventDispatcher = JSUTILS.EventDispatcher,
 24 		Event = JSUTILS.Event,
 25 		IOBoardEvent = BO.IOBoardEvent;
 26 
 27 	/**
 28 	 * Creates a new Stepper.
 29 	 *
 30 	 * @exports Stepper as BO.io.Stepper
 31 	 * @class Creates an interface to a Stepper motor. Use this object to set
 32 	 * the direction and number of steps for the motor to rotate. See
 33 	 * Breakout/examples/actuators/stepper.html for an example application.
 34 	 * @constructor
 35 	 * @param {IOBoard} board A reference to the IOBoard instance that the 
 36 	 * servo is attached to.
 37 	 * @param {Number} driverType. The type of driver (Stepper.DRIVER, 
 38 	 * Stepper.TWO_WIRE, or Stepper.FOUR_WIRE).
 39 	 * @param {Number} numStepsPerRev The number of steps to make 1 revolution. 
 40 	 * @param {Pin} directionPin If dirver interface, the pin used to control the direction.
 41 	 * If 2-wire interface, the 1st moter pin.
 42 	 * @param {Pin} stepPin If dirver interface, the pin used to control the steps.
 43 	 * If 2-wire interface, the 2nd moter pin.
 44 	 * @param {Pin} motorPin3 [optional] Only required for a 4-wire interface.
 45 	 * @param {Pin} motorPin4 [optional] Only required for a 4-wire interface.	 	 
 46 	 */
 47 	Stepper = function(board, driverType, numStepsPerRev, directionPin, stepPin, motorPin3, motorPin4) {
 48 		"use strict";
 49 		
 50 		// create a new id each time a new instance is created
 51 		this._id = instanceCounter++;
 52 		if (this._id > 5) {
 53 			console.log("Warning: A maximum of 6 Stepper instances can be created");
 54 			return;
 55 		}
 56 
 57 		this.name = "Stepper";
 58 		this._board = board;
 59 		this._evtDispatcher = new EventDispatcher(this);
 60 
 61 		var numStepsPerRevLSB = numStepsPerRev & 0x007F,
 62 			numStepsPerRevMSB = (numStepsPerRev >> 7) & 0x007F;
 63 
 64 		this._board.setDigitalPinMode(directionPin.number, Pin.DOUT);
 65 		this._board.setDigitalPinMode(stepPin.number, Pin.DOUT);
 66 
 67 		this._board.addEventListener(IOBoardEvent.SYSEX_MESSAGE, this.onSysExMessage.bind(this));
 68 		
 69 		switch (driverType) {
 70 			case Stepper.DRIVER:
 71 			case Stepper.TWO_WIRE:
 72 				// configure the stepper motor
 73 				this._board.sendSysex(Stepper.STEPPER,
 74 												[CONFIG,
 75 												this._id,
 76 												driverType,
 77 												numStepsPerRevLSB, 
 78 												numStepsPerRevMSB,
 79 												directionPin.number,
 80 												stepPin.number]);
 81 				break;
 82 			case Stepper.FOUR_WIRE:
 83 				this._board.setDigitalPinMode(motorPin3.number, Pin.DOUT);
 84 				this._board.setDigitalPinMode(motorPin4.number, Pin.DOUT);
 85 
 86 				// configure the stepper motor
 87 				this._board.sendSysex(Stepper.STEPPER,
 88 												[CONFIG,
 89 												this._id,
 90 												driverType,
 91 												numStepsPerRevLSB, 
 92 												numStepsPerRevMSB,
 93 												directionPin.number,
 94 												stepPin.number,
 95 												motorPin3.number,
 96 												motorPin4.number]);
 97 				break;
 98 		}
 99 
100 	};
101 
102 	Stepper.prototype = {
103 
104 		/**
105 		 * Number of steps in specified direction.
106 		 *
107 		 * @param {Number} numSteps The number of steps (max = +/-16,383).
108 		 * Positive value is clockwise, negative value is counter clockwise.
109 		 */
110 		step: function(numSteps) {
111 			var numStepsLSB = Math.abs(numSteps) & 0x007F,
112 				numStepsMSB = (Math.abs(numSteps) >> 7) & 0x007F,
113 				direction = Stepper.CLOCKWISE;		
114 
115 			if (numSteps > MAX_STEPS) {
116 				numSteps = MAX_STEPS;
117 				console.log("Warning: Maximum number of steps (16383) exceeded. Setting to step number to 16,383");
118 			}
119 			if (numSteps < -MAX_STEPS) {
120 				numSteps = -MAX_STEPS;
121 				console.log("Warning: Maximum number of steps (-16383) exceeded. Setting to step number to -16,383");
122 			}
123 
124 			if (numSteps > 0) {
125 				direction = Stepper.COUNTER_CLOCKWISE;
126 			}
127 
128 			this._board.sendSysex(Stepper.STEPPER, 
129 											[STEP,
130 											this._id,
131 											numStepsLSB,
132 											numStepsMSB,
133 											direction]);
134 		},
135 
136 		/**
137 		 * Set the stepper speed
138 		 *
139 		 * @param {Number} speed The spped in revolutions per minute (max = 16,384)
140 		 * (0 = counter clockwise, 1 = clockwise).
141 		 */
142 		setSpeed: function(speed) {
143 			var speedLSB = speed & 0x007F,
144 				speedMSB = (speed >> 7) & 0x007F;
145 
146 			if (speed > MAX_SPEED) {
147 				speed = MAX_SPEED;
148 				// TO DO: determin what the absolute max is when using stepper with Firmata.
149 				// It's likely far less that 16384 rpm
150 				console.log("Warning: Maximum speed (16,383) exceeded. Setting speed to 16,383 RPM");
151 			}					
152 
153 			this._board.sendSysex(Stepper.STEPPER, 
154 											[SPEED,
155 											this._id,
156 											speedLSB,
157 											speedMSB]);
158 		},			
159 
160 		/**
161 		 * Listen for stepping complete event
162 		 *
163 		 * @private
164 		 */
165 		onSysExMessage: function(event) {
166 			var message = event.message;
167 
168 			if (message[0] !== Stepper.STEPPER) {
169 				return;
170 			} else if (message[1] !== this._id) {
171 				return;
172 			} else {
173 				this.dispatchEvent(new Event(Event.COMPLETE));
174 			}
175 		},
176 
177 		/* implement EventDispatcher */
178 		
179 		/**
180 		 * @param {String} type The event type
181 		 * @param {Function} listener The function to be called when the event is fired
182 		 */
183 		addEventListener: function(type, listener) {
184 			this._evtDispatcher.addEventListener(type, listener);
185 		},
186 		
187 		/**
188 		 * @param {String} type The event type
189 		 * @param {Function} listener The function to be called when the event is fired
190 		 */
191 		removeEventListener: function(type, listener) {
192 			this._evtDispatcher.removeEventListener(type, listener);
193 		},
194 		
195 		/**
196 		 * @param {String} type The event type
197 		 * return {boolean} True is listener exists for this type, false if not.
198 		 */
199 		hasEventListener: function(type) {
200 			return this._evtDispatcher.hasEventListener(type);
201 		},
202 		
203 		/**
204 		 * @param {Event} type The Event object
205 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
206 		 * return {boolean} True if dispatch is successful, false if not.
207 		 */		
208 		dispatchEvent: function(event, optionalParams) {
209 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
210 		}			
211 	
212 	};
213 
214 	/** @constant */
215 	Stepper.STEPPER	= 0x72;
216 	/** @constant */
217 	Stepper.CLOCKWISE = 0;
218 	/** @constant */
219 	Stepper.COUNTER_CLOCKWISE = 1;
220 	/** @constant */
221 	Stepper.DRIVER = 1;
222 	/** @constant */
223 	Stepper.TWO_WIRE = 2;
224 	/** @constant */
225 	Stepper.FOUR_WIRE = 4;				
226 
227 	return Stepper;
228 
229 })();
