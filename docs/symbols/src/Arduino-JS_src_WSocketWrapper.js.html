  5 ARDJS.namespace('ARDJS.SocketEvent');
  6 
  7 ARDJS.SocketEvent = (function() {
  8 	
  9 	var SocketEvent;
 10 
 11 	// dependencies
 12 	var Event = ARDJS.Event;
 13 
 14 	/**
 15 	 * @exports SocketEvent as ARDJS.SocketEvent
 16 	 * @constructor
 17 	 * @augments ARDJS.Event
 18 	 * @param {String} type The event type
 19 	 */
 20 	SocketEvent = function(type) {
 21 		this.name = "SocketEvent"; // for testing
 22 		
 23 		// call the super class
 24 		// 2nd parameter is passed to EventDispatcher constructor
 25 		Event.call(this, type);		
 26 	};
 27 
 28 	// events
 29 	/** @constant */
 30 	SocketEvent.CONNECTED = "socketConnected";
 31 	/** @constant */
 32 	SocketEvent.MESSAGE = "socketMessage";
 33 	/** @constant */
 34 	SocketEvent.CLOSE = "socketClosed";
 35 
 36 	SocketEvent.prototype = ARDJS.inherit(Event.prototype);
 37 	SocketEvent.prototype.constructor = SocketEvent;	
 38 
 39 	return SocketEvent;
 40 
 41 }());

 44 /**
 45  * Wrapper for various client-side Websocket implementation to 
 46  * provide a consistent interface
 47  *
 48  * @author jeff hoefs
 49  */
 50 
 51 ARDJS.namespace('ARDJS.WSocketWrapper');
 52 
 53 ARDJS.WSocketWrapper = (function() {
 54 	"use strict";
 55 
 56  	var WSocketWrapper;
 57 
 58  	// dependencies
 59  	var EventDispatcher = ARDJS.EventDispatcher,
 60  		SocketEvent = ARDJS.SocketEvent;
 61 
 62 	/**
 63 	 * Creates a wrapper for various websocket implementations to unify the interface.
 64 	 *
 65 	 * @exports WSocketWrapper as ARDJS.WSocketWrapper
 66 	 * @constructor
 67 	 * @param {String} host The host address of the web server.
 68 	 * @param {Number} port The port to connect to on the web server.
 69 	 * @param {Boolean} useSocketIO Set true to use socket.io implementation, set false to use
 70 	 * native websocket implementation.
 71 	 * @param {String} protocol The websockt protocol definition (if necessary).
 72 	 */
 73 	WSocketWrapper = function(host, port, useSocketIO, protocol) {
 74 		this.name = "WSocketWrapper"; // for testing
 75 
 76 		EventDispatcher.call(this, this);
 77 
 78 		this._host = host;
 79 		this._port = port;
 80 		this._protocol = protocol || "default-protocol";
 81 		this._useSocketIO = useSocketIO || false;
 82 		this._socket = null;
 83 		this._readyState = ""; // only applies to native WebSocket implementations
 84 
 85 		this.init(this);
 86 
 87 	};
 88 
 89 	WSocketWrapper.prototype = ARDJS.inherit(EventDispatcher.prototype);
 90 	WSocketWrapper.prototype.constructor = WSocketWrapper;
 91 
 92 	/**
 93 	 * Initialize the websocket
 94 	 * @param {Object} self A reference to this websocket object.
 95 	 * @private
 96 	 */
 97 	WSocketWrapper.prototype.init = function(self) {
 98 
 99 		if (self._useSocketIO) {
100 			self._socket = io.connect("http://"+self._host+":"+self._port);
101 
102 			try {
103 				self._socket.on('connect', function() {
104 					self.dispatchEvent(new SocketEvent(SocketEvent.CONNECTED));
105 
106 					self._socket.on('message', function(msg) {
107 						self.dispatchEvent(new SocketEvent(SocketEvent.MESSAGE), {message: msg});
108 					});
109 				});
110 				
111 			} catch(exception){
112 				console.log("Error "+exception);
113 			}
114 
115 		} else {
116 			
117 			try {
118 
119 				if ("MozWebSocket" in window) {
120 					self._socket = new MozWebSocket("ws://"+self._host+":"+self._port, self._protocol);
121 				} else if ("WebSocket" in window) {
122 					// Safari doesn't like protocol parameter
123 					//self._socket = new WebSocket("ws://"+self._host+":"+self._port, self._protocol);
124 					self._socket = new WebSocket("ws://"+self._host+":"+self._port);
125 				} else {
126 					console.log("Websockets not supported by this browser");
127 					throw "Websockets not supported by this browser";
128 					return;	
129 				}
130 
131 				console.log("Starting up...");
132 
133 				self._socket.onopen = function() {
134 
135 					self.dispatchEvent(new SocketEvent(SocketEvent.CONNECTED));
136 
137 					self._socket.onmessage = function(msg) {
138 						self.dispatchEvent(new SocketEvent(SocketEvent.MESSAGE), {message: msg.data});
139 					};
140 
141 					self._socket.onclose = function() {
142 						self._readyState = self._socket.readyState;
143 						self.dispatchEvent(new SocketEvent(SocketEvent.CLOSE));	
144 					};
145 
146 				};
147 				
148 			} catch(exception){
149 				console.log("Error "+exception);
150 			}
151 				
152 		}
153 
154 	};
155 
156 	/**
157 	 * Send a message
158 	 * @param {String} message The message to send
159 	 */
160 	WSocketWrapper.prototype.send = function(message) {
161 		// to do: ensure socket is not null before trying to send
162 		this._socket.send(message);
163 	};
164 
165 	// to do: ensure socket is not null before trying to get readyState
166 
167 	/**
168 	 * [read-only] Wrapper for the readyState method of the native websocket implementation
169 	 * <p>CONNECTING = 0, OPEN = 1, CLOSING = 2, CLOSED = 3</p>
170 	 * @name WSocketWrapper#readyState
171 	 * @property
172 	 * @type String
173 	 */		 
174 	WSocketWrapper.prototype.__defineGetter__("readyState", function() { return this._readyState; });
175 
176 	return WSocketWrapper;
177 
178 }());
