  6 BREAKOUT.namespace('BREAKOUT.io.Servo');
  7 
  8 BREAKOUT.io.Servo = (function() {
  9 
 10 	var Servo;
 11 
 12 	// dependencies
 13 	var Pin = BREAKOUT.Pin;
 14 
 15 	/**
 16 	 * Creates a new Servo
 17 	 *
 18 	 * @exports Servo as BREAKOUT.io.Servo
 19 	 * @constructor
 20 	 * @param {IOBoard} board A reference to the IOBoard instance that the servo is attached to.
 21 	 * @param {Pin} servoPin A reference to the Pin the servo is connected to.
 22 	 * @param {Number} minAngle The minimum angle the server can rotate to (default = 0).
 23 	 * @param {Number} maxAngle The maximum angle the server can rotate to (default = 180).
 24 	 */
 25 	Servo = function(board, servoPin, minAngle, maxAngle) {
 26 		"use strict";
 27 		
 28 		this.name = "Servo";
 29 
 30 		this._pin = servoPin;
 31 		this._angle;
 32 		this._minAngle = minAngle || 0;
 33 		this._maxAngle = maxAngle || 180;
 34 
 35 		var pinNumber = servoPin.number;
 36 		
 37 		// sendServoAttach will set the pin mode to Pin.SERVO
 38 		board.sendServoAttach(pinNumber);
 39 	}
 40 
 41 	Servo.prototype = {
 42 
 43 		/**
 44 		 * Set the angle (in degrees) to rotate the server head to.
 45 		 * @name Servo#angle
 46 		 * @property
 47 		 * @type Number
 48 		 */ 
 49 		set angle(value) {
 50 			if (this._pin.getType() == Pin.SERVO) {
 51 				this._angle = value;
 52 				//this._pin.value = this._angle;
 53 				this._pin.value = Math.max(0, Math.min(1, (this._angle - this._minAngle) / 
 54 								(this._maxAngle - this._minAngle) * Servo.COEF_TO_0_180));
 55 
 56 			}
 57 		},
 58 		get angle() {
 59 			if (this._pin.getType() == Pin.SERVO) {
 60 				return this._angle;
 61 			}
 62 		}		
 63 	};
 64 
 65 	/**
 66 	 * The scale to convert 0-1 (0-255 in 8bit) to 0-0.706 (0-180 in 8bit).
 67 	 */
 68 	Servo.COEF_TO_0_180 = 180 / 255;
 69 
 70 	return Servo;
 71 
 72 }());
