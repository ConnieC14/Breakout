  5 BREAKOUT.namespace('BREAKOUT.Pin');
  6 
  7 BREAKOUT.Pin = (function() {
  8 	//"use strict";	// closure compiler doesn't like this... should be fixed soon though
  9 
 10 	var Pin;
 11 
 12 	// dependencies
 13 	var EventDispatcher = BREAKOUT.EventDispatcher,
 14 		Event = BREAKOUT.Event;
 15 
 16 	/**
 17 	 * An object to represent an IOBoard pin
 18 	 *
 19 	 * @exports Pin as BREAKOUT.Pin
 20 	 * @constructor
 21 	 * @param {Number} number The pin number
 22 	 * @param {Number} type The type of pin
 23 	 */
 24 	Pin = function(number, type) {
 25 
 26 		this.name = "Pin"; // for testing
 27 
 28 		this._type = type;
 29 		this._capabilities;
 30 		this._number = number;
 31 		this._analogNumber = undefined;
 32 		this._value = -1;
 33 		this._lastValue = -1;
 34 		this._preFilterValue;
 35 		this._average = 0;
 36 		this._minimum = Math.pow(2, 16);
 37 		this._maximum = 0;
 38 		this._avg = 0;
 39 		this._sum = 0;
 40 		this._numSamples = 0;
 41 		
 42 		this._evtDispatcher = new EventDispatcher(this);	
 43 
 44 	};
 45 
 46 	Pin.prototype = {
 47 		
 48 		/**
 49 		 * The analogNumber sould only be set internally.
 50 		 * @private
 51 		 */		
 52 		setAnalogNumber: function(num) {
 53 			this._analogNumber = num;
 54 		},
 55 
 56 		/**
 57 		 * [read-only] The analog pin number used by the IOBoard (printed on board or datasheet).
 58 		 * @name Pin#analogNumber
 59 		 * @property
 60 		 * @type Number
 61 		 */	
 62 		get analogNumber() {
 63 			return this._analogNumber;
 64 		},		
 65 
 66 		/**
 67 		 * [read-only] The pin number corresponding to the Arduino documentation for the type of board.
 68 		 * @name Pin#number
 69 		 * @property
 70 		 * @type Number
 71 		 */			 
 72 		get number() {
 73 			return this._number;
 74 		},
 75 		
 76 		/**
 77 		 * [read-only] The average value of the pin over time. Call clear() to reset.
 78 		 * @name Pin#average
 79 		 * @property
 80 		 * @type Number
 81 		 */			 
 82 		get average() {
 83 			return this._average;
 84 		},
 85 
 86 		/**
 87 		 * [read-only] The minimum value of the pin over time. Call clear() to reset.
 88 		 * @name Pin#minimum
 89 		 * @property
 90 		 * @type Number
 91 		 */			 	
 92 		get minimum() {
 93 			return this._minimum;
 94 		},
 95 		
 96 		/**
 97 		 * [read-only] The maximum value of the pin over time. Call clear() to reset.
 98 		 * @name Pin#maximum
 99 		 * @property
100 		 * @type Number
101 		 */			 
102 		get maximum() {
103 			return this._maximum;
104 		},
105 		
106 		/**
107 		 * The current digital or analog value of the pin.
108 		 * @name Pin#value
109 		 * @property
110 		 * @type Number
111 		 */		 
112 		get value() {
113 			return this._value;
114 		},
115 		set value(val) {
116 			this.calculateMinMaxAndMean(val);
117 			this._lastValue = this._value;
118 			this._preFilterValue = val;
119 			this._value = val;
120 			this.detectChange(this._lastValue, this._value);
121 		},
122 		
123 		/**
124 		 * [read-only] The last pin value.
125 		 * @name Pin#lastValue
126 		 * @property
127 		 * @type Number
128 		 */			 
129 		get lastValue() {
130 			return this._lastValue;
131 		},
132 		
133 		/**
134 		 * [read-only] The value before any filters were applied.
135 		 * @name Pin#preFilterValue
136 		 * @property
137 		 * @type Number
138 		 */			 
139 		get preFilterValue() {
140 			return this._preFilterValue;
141 		},
142 
143 		/**
144 		 * The type/mode of the pin (eg. Pin.DIN, Pin.DOUT, etc). Use IOBoard.setDigitalPinMode(pinNumber) 
145 		 * to set the pin type.
146 		 * @return {Number} The pin type/mode
147 		 */	
148 		getType: function() {
149 			return this._type;
150 		},
151 
152 		/**
153 		 * Set the pin type. This method should only be used internaly.
154 		 * @private
155 		 */
156 		setType: function(pinType) {
157 			// ensure pin type is valid
158 			if (pinType >= 0 && pinType < Pin.TOTAL_PIN_MODES) {
159 				this._type = pinType;
160 			}
161 		},			
162 
163 		/**
164 		 * An object storing the capabilities of the pin.
165 		 * @return {Object} An object describing the capabilities of this Pin.
166 		 */	
167 		getCapabilities: function() {
168 			return this._capabilities;
169 		},
170 
171 		/**
172 		 * This method should only be used internally.
173 		 * @private
174 		 */
175 		setCapabilities: function(pinCapabilities) {
176 			this._capabilities = pinCapabilities;
177 		},		
178 
179 		/**
180 		 * Dispatch a Change event whenever a pin value changes
181 		 * @private
182 		 */
183 		detectChange: function(oldValue, newValue) {
184 			if (oldValue == newValue) return;
185 			//console.log("detect change");
186 			this.dispatchEvent(new Event(Event.CHANGE));
187 		},
188 		
189 		/**
190 		 * From funnel Pin.as
191 		 * @private
192 		 */
193 		clearWeight: function() {
194 			this._sum = this._average;
195 			this._numSamples = 1;
196 		},
197 		
198 		/**
199 		 * From funnel Pin.as
200 		 * @private
201 		 */
202 		calculateMinMaxAndMean: function(val) {
203 			var MAX_SAMPLES = Number.MAX_VALUE;
204 
205 			this._minimum = Math.min(val, this._minimum);
206 			this._maximum = Math.max(val, this._maximum);
207 			
208 			this._sum += val;
209 			this._average = this._sum / (++this._numSamples);
210 			if (this._numSamples >= MAX_SAMPLES) {
211 				this.clearWeight();
212 			}
213 		},
214 			
215 		/**
216 		 * Resets the minimum, maximum, average and lastValue of the pin.
217 		 */
218 		clear: function() {
219 			this._minimum = this._maximum = this._average = this._lastValue = this._preFilterValue;
220 			this.clearWeight();
221 		},		
222 
223 
224 		/* implement EventDispatcher */
225 		
226 		/**
227 		 * @param {String} type The event type
228 		 * @param {Function} listener The function to be called when the event is fired
229 		 */
230 		addEventListener: function(type, listener) {
231 			this._evtDispatcher.addEventListener(type, listener);
232 		},
233 		
234 		/**
235 		 * @param {String} type The event type
236 		 * @param {Function} listener The function to be called when the event is fired
237 		 */
238 		removeEventListener: function(type, listener) {
239 			this._evtDispatcher.removeEventListener(type, listener);
240 		},
241 		
242 		/**
243 		 * @param {String} type The event type
244 		 * return {boolean} True is listener exists for this type, false if not.
245 		 */
246 		hasEventListener: function(type) {
247 			return this._evtDispatcher.hasEventListener(type);
248 		},
249 
250 		/**
251 		 * @param {Event} type The Event object
252 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
253 		 * return {boolean} True if dispatch is successful, false if not.
254 		 */	
255 		dispatchEvent: function(event, optionalParams) {
256 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
257 		}		
258 			
259 	};
260 
261 	/** @constant */
262 	Pin.HIGH = 1;
263 	/** @constant */
264 	Pin.LOW	= 0;
265 	/** @constant */
266 	Pin.ON = 1;
267 	/** @constant */
268 	Pin.OFF = 0;
269 
270 	// pin modes
271 	/** @constant */
272 	Pin.DIN = 0x00;
273 	/** @constant */
274 	Pin.DOUT = 0x01;
275 	/** @constant */
276 	Pin.AIN = 0x02;
277 	/** @constant */
278 	Pin.AOUT = 0x03;
279 	/** @constant */
280 	Pin.PWM = 0x03;
281 	/** @constant */
282 	Pin.SERVO = 0x04;
283 	/** @constant */
284 	Pin.SHIFT = 0x05;
285 	/** @constant */
286 	Pin.I2C = 0x06;
287 	/** @constant */
288 	Pin.TOTAL_PIN_MODES = 7;
289 
290 	return Pin;
291 
292 }());
