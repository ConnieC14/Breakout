  6 JSUTILS.namespace('BREAKOUT.io.Potentiometer');
  7 
  8 BREAKOUT.io.Potentiometer = (function() {
  9 
 10 	var Potentiometer;
 11 
 12 	// dependencies
 13 	var Pin = BREAKOUT.Pin,
 14 		PhysicalInputBase = BREAKOUT.PhysicalInputBase,
 15 		Scaler = BREAKOUT.filters.Scaler,
 16 		Convolution = BREAKOUT.filters.Convolution,
 17 		Event = JSUTILS.Event;
 18 
 19 	/**
 20 	 * Creates a new Potentiometer. Listen for a change event (Event.CHANGE).
 21 	 *
 22 	 * @exports Potentiometer as BREAKOUT.io.Potentiometer
 23 	 * @class
 24 	 * @augments BREAKOUT.PhysicalInputBase	 
 25 	 * @param {IOBoard} board A reference to the IOBoard instance that the servo is attached to.
 26 	 * @param {Pin} pin A reference to the Pin the potentiometer is connected to.
 27 	 * @param {Boolean} enableSmoothing True to enable smoothing, false to disable. Default is false.
 28 	 */
 29 	Potentiometer = function(board, pin, enableSmoothing) {
 30 		"use strict";
 31 
 32 		PhysicalInputBase.call(this);
 33 		
 34 		this.name = "Potentiometer";
 35 		this._pin = pin;
 36 
 37 		enableSmoothing = enableSmoothing || false;
 38 
 39 		var analogPinNumber = this._pin.analogNumber;
 40 		board.enableAnalogPin(analogPinNumber);
 41 
 42 		if (enableSmoothing) {
 43 			this._pin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
 44 		}
 45 
 46 		this._pin.addEventListener(Event.CHANGE, this.onPinChange.bind(this));
 47 	};
 48 
 49 	Potentiometer.prototype = JSUTILS.inherit(PhysicalInputBase.prototype);
 50 	Potentiometer.prototype.constructor = Potentiometer;
 51 	
 52 	/**
 53 	 * [read-only] The current value of the potentiometer.
 54 	 * @name Potentiometer#value
 55 	 * @property
 56 	 * @type Number
 57 	 */ 
 58 	Potentiometer.prototype.__defineGetter__("value", function() { return this._pin.value; });
 59 
 60 	/**
 61 	 * [read-only] Get the (pre-filtered) average value of the potentiometer
 62 	 * @name Potentiometer#average
 63 	 * @property
 64 	 * @type Number
 65 	 */ 
 66 	Potentiometer.prototype.__defineGetter__("average", function() { return this._pin.average; });
 67 
 68 	/**
 69 	 * [read-only] Get the value of the potentiometer before filters are applied.
 70 	 * @name Potentiometer#preFilterValue
 71 	 * @property
 72 	 * @type Number
 73 	 */ 
 74 	Potentiometer.prototype.__defineGetter__("preFilterValue", function() { return this._pin.preFilterValue; });
 75 
 76 	/**
 77 	 * [read-only] Get the (pre-filtered) minimum value read by the potentiometer.
 78 	 * @name Potentiometer#minimum
 79 	 * @property
 80 	 * @type Number
 81 	 */ 
 82 	Potentiometer.prototype.__defineGetter__("minimum", function() { return this._pin.minimum; });
 83 
 84 	/**
 85 	 * [read-only] Get the (pre-filtered) maximum value read by the potentiometer.
 86 	 * @name Potentiometer#maximum
 87 	 * @property
 88 	 * @type Number
 89 	 */ 
 90 	Potentiometer.prototype.__defineGetter__("maximum", function() { return this._pin.maximum; });
 91 
 92 	/**
 93 	 * Resets the minimum, maximum and average values.
 94 	 */
 95 	Potentiometer.prototype.clear = function() {
 96 		this._pin.clear();
 97 	};
 98 
 99 	/**
100 	 * Scale from the minimum and maximum input values to 0.0 -> 1.0. This is
101 	 * useful for sensors such as a flex sensor that may not return the full
102 	 * range of 0 to 1. 
103 	 *
104 	 * @param {Number} minimum The new minimum range (must be less than the maximum).
105 	 * @param {Number} maximum The new maximum range.
106 	 */
107 	Potentiometer.prototype.setRange = function(minimum, maximum) {
108 		minimum = minimum || 0;
109 		maximum = maximum || 1;
110 		this._pin.addFilter(new Scaler(minimum, maximum, 0, 1, Scaler.LINEAR));
111 	};
112 
113 	/**
114 	 * @private
115 	 */
116 	Potentiometer.prototype.onPinChange = function(event) {
117 		this.dispatchEvent(new Event(Event.CHANGE));
118 	};
119 
120 	// document events
121 
122 	/**
123 	 * The change event is dispatched when the potentiometer value changes.
124 	 * @name Potentiometer#change
125 	 * @type JSUTILS.Event.CHANGE
126 	 * @event
127 	 * @example
128 	 * pot.addEventListener(Event.CHANGE, onValueChange);
129 	 *
130 	 * function onValueChange(event) {
131 	 *   console.log("value = " + event.target.value);	
132 	 * }
133 	 * @param {BREAKOUT.Potentiometer} target A reference to the Potentiometer object
134 	 */
135 
136 	return Potentiometer;
137 
138 }());
