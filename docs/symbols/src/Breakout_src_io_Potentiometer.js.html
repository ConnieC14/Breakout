1 /**

  4 
  5 BREAKOUT.namespace('BREAKOUT.io.Potentiometer');
  6 
  7 BREAKOUT.io.Potentiometer = (function() {
  8 
  9 	var Potentiometer;
 10 
 11 	// dependencies
 12 	var Pin = BREAKOUT.Pin,
 13 		PhysicalInputBase = BREAKOUT.PhysicalInputBase,
 14 		Scaler = BREAKOUT.filters.Scaler,
 15 		Convolution = BREAKOUT.filters.Convolution,
 16 		Event = BREAKOUT.Event;
 17 
 18 	/**
 19 	 * Creates a new Potentiometer. Listen for a change event (Event.CHANGE).
 20 	 *
 21 	 * @exports Potentiometer as BREAKOUT.io.Potentiometer
 22 	 * @constructor
 23 	 * @param {IOBoard} board A reference to the IOBoard instance that the servo is attached to.
 24 	 * @param {Pin} pin A reference to the Pin the potentiometer is connected to.
 25 	 * @param {Boolean} enableSmoothing True to enable smoothing, false to disable. Default is false.
 26 	 */
 27 	Potentiometer = function(board, pin, enableSmoothing) {
 28 		"use strict";
 29 
 30 		PhysicalInputBase.call(this);
 31 		
 32 		this.name = "Potentiometer";
 33 		this._pin = pin;
 34 
 35 		enableSmoothing = enableSmoothing || false;
 36 
 37 		var analogPinNumber = this._pin.analogNumber;
 38 		board.enableAnalogPin(analogPinNumber);
 39 
 40 		if (enableSmoothing) {
 41 			this._pin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
 42 		}
 43 
 44 		this._pin.addEventListener(Event.CHANGE, this.onPinChange.bind(this));
 45 	}
 46 
 47 	Potentiometer.prototype = BREAKOUT.inherit(PhysicalInputBase.prototype);
 48 	Potentiometer.prototype.constructor = Potentiometer;
 49 	
 50 	/**
 51 	 * [read-only] The current value of the potentiometer.
 52 	 * @name Potentiometer#value
 53 	 * @property
 54 	 * @type Number
 55 	 */ 
 56 	Potentiometer.prototype.__defineGetter__("value", function() { return this._pin.value; });
 57 
 58 	/**
 59 	 * [read-only] Get the (pre-filtered) average value of the potentiometer
 60 	 * @name Potentiometer#average
 61 	 * @property
 62 	 * @type Number
 63 	 */ 
 64 	Potentiometer.prototype.__defineGetter__("average", function() { return this._pin.average; });
 65 
 66 	/**
 67 	 * [read-only] Get the value of the potentiometer before filters are applied.
 68 	 * @name Potentiometer#preFilterValue
 69 	 * @property
 70 	 * @type Number
 71 	 */ 
 72 	Potentiometer.prototype.__defineGetter__("preFilterValue", function() { return this._pin.preFilterValue; });
 73 
 74 	/**
 75 	 * [read-only] Get the (pre-filtered) minimum value read by the potentiometer.
 76 	 * @name Potentiometer#minimum
 77 	 * @property
 78 	 * @type Number
 79 	 */ 
 80 	Potentiometer.prototype.__defineGetter__("minimum", function() { return this._pin.minimum; });
 81 
 82 	/**
 83 	 * [read-only] Get the (pre-filtered) maximum value read by the potentiometer.
 84 	 * @name Potentiometer#maximum
 85 	 * @property
 86 	 * @type Number
 87 	 */ 
 88 	Potentiometer.prototype.__defineGetter__("maximum", function() { return this._pin.maximum; });
 89 
 90 	/**
 91 	 * Resets the minimum, maximum and average values.
 92 	 */
 93 	Potentiometer.prototype.clear = function() {
 94 		this._pin.clear();
 95 	};
 96 
 97 	/**
 98 	 * Scale the potentiometer range (default 0.0 to 1.0) to the new specified range.
 99 	 * <p>For example, setting the setRange(0, 180) will scale the potentiometer
100 	 * values from 0 to 180.</p>
101 	 *
102 	 * @param {Number} minimum The new minimum range (must be less than the maximum).
103 	 * @param {Number} maximum The new maximum range.
104 	 */
105 	Potentiometer.prototype.setRange = function(minimum, maximum) {
106 		minimum = minimum || 0;
107 		maximum = maximum || 1;
108 		//this._pin.filters = [new Scaler(0, 1, minimum, maximum, Scaler.LINEAR)];
109 		this._pin.addFilter(new Scaler(0, 1, minimum, maximum, Scaler.LINEAR));
110 	};
111 
112 	/**
113 	 * @private
114 	 */
115 	Potentiometer.prototype.onPinChange = function(event) {
116 		this.dispatchEvent(new Event(Event.CHANGE));
117 	};
118 	
119 
120 	return Potentiometer;
121 
122 }());
