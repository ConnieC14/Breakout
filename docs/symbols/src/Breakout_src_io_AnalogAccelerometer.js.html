 12 BREAKOUT.namespace('BREAKOUT.io.AnalogAccelerometer');

 14 BREAKOUT.io.AnalogAccelerometer = (function() {
 15 	"use strict";
 16 
 17 	var AnalogAccelerometer;
 18 
 19 		// dependencies
 20 	var PhysicalInputBase = BREAKOUT.PhysicalInputBase,
 21 		Event = BREAKOUT.Event,
 22 		Scaler = BREAKOUT.filters.Scaler,
 23 		Convolution = BREAKOUT.filters.Convolution;
 24 
 25 	/**
 26 	 * Creates a new Analog Accelerometer object
 27 	 *
 28 	 * @exports AnalogAccelerometer as BREAKOUT.io.AnalogAccelerometer
 29 	 * @constructor
 30 	 * @param {IOBoard} board A reference to the IOBoard instance
 31 	 * @param {Pin} xPin A reference to the Pin connected to the x axis of the accelerometer
 32 	 * @param {Pin} yPin A reference to the Pin connected to the y axis of the accelerometer
 33 	 * @param {Pin} zPin A reference to the Pin connected to the z axis of the accelerometer
 34 	 * @param {Number} dynamicRange The range of the acceleromter in Gs (typically 2 or 3 for an 
 35 	 * analog accelerometer). See the datasheet for the acceleromter to get the exact value.
 36 	 * @param {Boolean} enableSmoothing True to enable smoothing, false to disable. Default is false.
 37 	 */
 38 	AnalogAccelerometer = function(board, xPin, yPin, zPin, dynamicRange, enableSmoothing) {
 39 
 40 		// call the super class
 41 		PhysicalInputBase.call(this);
 42 
 43 		this.name = "AnalogAccelerometer";
 44 
 45 		// enable the analog pins
 46 		board.enableAnalogPin(xPin.analogNumber);
 47 		board.enableAnalogPin(yPin.analogNumber);
 48 		board.enableAnalogPin(zPin.analogNumber);		
 49 
 50 		this._enableSmoothing = enableSmoothing || false;
 51 		this._xPin = xPin || null;
 52 		this._yPin = yPin || null;
 53 		this._zPin = zPin || null;
 54 
 55 		// common accelerometer interface values:
 56 		this._dynamicRange = dynamicRange || 1;
 57 		this._x = 0;
 58 		this._y = 0;
 59 		this._z = 0;
 60 
 61 		if (this._xPin !== null) {
 62 			this._xPin.addEventListener(Event.CHANGE, this.xAxisChanged.bind(this));
 63 		}
 64 
 65 		if (this._yPin !== null) {
 66 			this._yPin.addEventListener(Event.CHANGE, this.yAxisChanged.bind(this));
 67 		}
 68 		
 69 		if (this._zPin !== null) {
 70 			this._zPin.addEventListener(Event.CHANGE, this.zAxisChanged.bind(this));
 71 		}
 72 		
 73 	};
 74 
 75 	AnalogAccelerometer.prototype = BREAKOUT.inherit(PhysicalInputBase.prototype);
 76 	AnalogAccelerometer.prototype.constructor = AnalogAccelerometer;
 77 
 78 	// Implement Acceleromter interface:
 79 
 80 	/**
 81 	 * [read-only] The current range setting of the accelerometer in units 
 82 	 * of gravity (9.8 m/sec2).
 83 	 * @name AnalogAccelerometer#dynamicRange
 84 	 * @property
 85 	 * @type Number
 86 	 */ 
 87 	AnalogAccelerometer.prototype.__defineGetter__("dynamicRange", function() { return this._dynamicRange; });
 88 
 89 	/**
 90 	 * [read-only] The x axis of the accelerometer in units 
 91 	 * of gravity (9.8 m/sec2).
 92 	 * @name AnalogAccelerometer#x
 93 	 * @property
 94 	 * @type Number
 95 	 */ 
 96 	AnalogAccelerometer.prototype.__defineGetter__("x", function() { return this._x; });
 97 
 98 	/**
 99 	 * [read-only] The y axis of the accelerometer in units 
100 	 * of gravity (9.8 m/sec2).
101 	 * @name AnalogAccelerometer#y
102 	 * @property
103 	 * @type Number
104 	 */ 
105 	AnalogAccelerometer.prototype.__defineGetter__("y", function() { return this._y; });
106 
107 	/**
108 	 * [read-only] The z axis of the accelerometer in units 
109 	 * of gravity (9.8 m/sec2).
110 	 * @name AnalogAccelerometer#z
111 	 * @property
112 	 * @type Number
113 	 */ 
114 	AnalogAccelerometer.prototype.__defineGetter__("z", function() { return this._z; });
115 
116 
117 	// Methods specific to this Accelerometer type:
118 
119 	AnalogAccelerometer.prototype.__defineGetter__("xPin", function() { 
120 		return this._xPin;
121 	});
122 
123 	AnalogAccelerometer.prototype.__defineGetter__("yPin", function() { 
124 		return this._yPin;
125 	});	
126 	
127 	AnalogAccelerometer.prototype.__defineGetter__("zPin", function() { 
128 		return this._zPin;
129 	});			
130 
131 	/**
132 	 * [read-only] The rotation on the x axis 
133 	 * of gravity (9.8 m/sec2).
134 	 * @name AnalogAccelerometer#rotationX
135 	 * @property
136 	 * @type Number
137 	 */ 
138 	AnalogAccelerometer.prototype.__defineGetter__("rotationX", function() { 
139 		var sinX = Math.min(Math.max(this._x, -1), 1);
140 		return Math.asin(sinX) / Math.PI * 180;
141 	});	
142 
143 	/**
144 	 * [read-only] The rotation on the y axis 
145 	 * of gravity (9.8 m/sec2).
146 	 * @name AnalogAccelerometer#rotationY
147 	 * @property
148 	 * @type Number
149 	 */ 
150 	AnalogAccelerometer.prototype.__defineGetter__("rotationY", function() { 
151 		var sinY = Math.min(Math.max(this._y, -1), 1);
152 		return Math.asin(sinY) / Math.PI * 180;
153 	});	
154 
155 	/**
156 	 * [read-only] The rotation on the z axis 
157 	 * of gravity (9.8 m/sec2).
158 	 * @name AnalogAccelerometer#rotationZ
159 	 * @property
160 	 * @type Number
161 	 */ 
162 	AnalogAccelerometer.prototype.__defineGetter__("rotationZ", function() { 
163 		var sinZ = Math.min(Math.max(this._z, -1), 1);
164 		return Math.asin(sinZ) / Math.PI * 180;
165 	});		
166 
167 	/**
168 	 * Scale the range for the specified axis (from 0 to 1) to (minimum to maximum).
169 	 * 
170 	 * @param axis the axis to set new range (AnalogAccelerometer.X_AXIS, 
171 	 * AnalogAccelerometer.Y_AXIS or AnalogAccelerometer.Z_AXIS).
172 	 * @param {Number} minimum The new minimum value
173 	 * @param {Number} maximum The new maximum value
174 	 */
175 	AnalogAccelerometer.prototype.setRangeFor = function(axis, minimum, maximum) {
176 		var range = this._dynamicRange;
177 
178 		if (axis === AnalogAccelerometer.X_AXIS) {
179 			if (this._xPin !== null) {
180 				this._xPin.filters = [new Scaler(minimum, maximum, -range, range, Scaler.LINEAR)];
181 				if (this._enableSmoothing) {
182 					this._xPin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
183 				}
184 			}
185 		} else if (axis === AnalogAccelerometer.Y_AXIS) {
186 			if (this._yPin !== null) {
187 				this._yPin.filters = [new Scaler(minimum, maximum, -range, range, Scaler.LINEAR)];
188 				if (this._enableSmoothing) {
189 					this._yPin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
190 				}
191 			}
192 		} else if (axis === AnalogAccelerometer.Z_AXIS) {
193 			if (this._zPin !== null) {
194 				this._zPin.filters = [new Scaler(minimum, maximum, -range, range, Scaler.LINEAR)];
195 				if (this._enableSmoothing) {
196 					this._zPin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
197 				}
198 			}
199 		}
200 	};
201 
202 	// calibration:
203 	// for each axis, calculate variance from -1 and 1
204 	// assume zero g = supply voltage/2 and sensitivity is supply voltage / 10 per g
205 	// at -1 g, voltage should be (supply voltage/2) - (supply voltage / 10)
206 	// at 1 g, voltage should be (supply voltage/2) + (supply voltage / 10)
207 
208 	/**
209 	 * Use this method to get the minimum and maximum range values for an axis. Create a new object
210 	 * to store the return value and then pass obj.min and obj.max along with the respective axis
211 	 * identifier to the setRangeFor method.
212 	 * 
213 	 * @param {Number} minVoltage The minimum value reported on the axis
214 	 * @param {Number} maxVoltage The maximum value reported on the axis
215 	 * @param {Number} supplyVoltage The supply voltage of the Acceleromter (enter as 3.3, 3.0, 5.0, etc).
216 	 * @return {Object} An object containing the min and max range values to be passed to the 
217 	 * setRangeFor method.
218 	 */
219 	AnalogAccelerometer.prototype.getCalibratedRange = function(minVoltage, maxVoltage, supplyVoltage) {
220 		var range = {min:0, max:0};
221 		
222 		var mVPerG = (maxVoltage - minVoltage) / 2;
223 		
224 		// find zero G (average of min and max)
225 		var zeroG = (minVoltage + maxVoltage) / 2;
226 		
227 		range.min = (zeroG - (mVPerG * this._dynamicRange))/supplyVoltage;
228 		range.max = (zeroG + (mVPerG * this._dynamicRange))/supplyVoltage;
229 		
230 		return range;
231 	};		
232 
233 	/**
234 	 * @private
235 	 */
236 	AnalogAccelerometer.prototype.xAxisChanged = function(event) {
237 		this._x = event.target.value;
238 		this.dispatchEvent(new Event(Event.CHANGE));
239 	};
240 
241 	/**
242 	 * @private
243 	 */
244 	AnalogAccelerometer.prototype.yAxisChanged = function(event) {
245 		this._y = event.target.value;
246 		this.dispatchEvent(new Event(Event.CHANGE));
247 	};
248 	
249 	/**
250 	 * @private
251 	 */
252 	AnalogAccelerometer.prototype.zAxisChanged = function(event) {
253 		this._z = event.target.value;
254 		this.dispatchEvent(new Event(Event.CHANGE));
255 	};		
256 
257 	/** @constant */
258 	AnalogAccelerometer.X_AXIS = 0;
259 	/** @constant */
260 	AnalogAccelerometer.Y_AXIS = 1;
261 	/** @constant */
262 	AnalogAccelerometer.Z_AXIS = 2;	
263 
264 	return AnalogAccelerometer;
265 
266 }());
