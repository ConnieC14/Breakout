  6 JSUTILS.namespace('JSUTILS.Event');
  7 
  8 JSUTILS.Event = (function() {
  9 
 10 	var Event;
 11 
 12 	/** 
 13 	 * Event 'base class' (but it can also be instantiated directly)
 14 	 *
 15 	 * @exports Event as JSUTILS.Event
 16 	 * @constructor
 17 	 * @param {String} type event type
 18 	 */
 19 	Event = function(type) {
 20 		/** @property {String} type The event type. */
 21 		this.type = type;
 22 		/** @property {Object} target The event target. */
 23 		this.target = null;
 24 
 25 		this.name = "Event"; // for testing
 26 	};
 27 
 28 	/** @constant */
 29 	Event.CONNECTED = "connected";
 30 	/** @constant */
 31 	Event.CHANGE	= "change";
 32 	/** @constant */
 33 	Event.COMPLETE	= "complete";
 34 
 35 	return Event;
 36 
 37 }());
