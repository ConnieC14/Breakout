1 /**
  2  * Creates and BlinkM object
  3  *
  4  * @constructor
  5  * @param {Arduino} board The Arduino instance
  6  * @param {Number} address The i2c address of the BlinkM module
  7  */
  8 function BlinkM(board, address) {
  9 
 10 	address = address || 0x09;	// default i2c address for BlinkM
 11 	
 12 	var self = this;
 13 
 14 	// call super class
 15 	I2CBase.call(self, board, address);
 16 	
 17 	this.goToRGBColorNow = function(color) {
 18 		self.sendI2CRequest([I2CBase.WRITE, address, 0x6E, color[0], color[1], color[2]]);
 19 	}
 20 
 21 	this.fadeToRGBColor = function(color, speed) {
 22 		speed = speed || -1;
 23 		if (speed >= 0) {
 24 			self.sendI2CRequest([I2CBase.WRITE, address, 0x66, speed]);
 25 		}
 26 		self.sendI2CRequest([I2CBase.WRITE, address, 0x63, color[0], color[1], color[2]]);
 27 	}
 28 
 29 	this.fadeToRandomRGBColor = function (color, speed) {
 30 		speed = speed || -1;
 31 		if (speed >= 0) {
 32 			self.sendI2CRequest([I2CBase.WRITE, address, 0x66, speed]);
 33 		}
 34 		self.sendI2CRequest([I2CBase.WRITE, address, 0x43, color[0], color[1], color[2]]);
 35 	}
 36 
 37 	this.fadeToHSBColor = function(color, speed) {
 38 		speed = speed || -1;
 39 		if (speed >= 0) {
 40 			self.sendI2CRequest([I2CBase.WRITE, address, 0x66, speed]);
 41 		}
 42 		self.sendI2CRequest([I2CBase.WRITE, address, 0x68, color[0], color[1], color[2]]);
 43 	}
 44 
 45 	this.fadeToRandomHSBColor = function(color, speed) {
 46 		speed = speed || -1;
 47 		if (speed >= 0) {
 48 			self.sendI2CRequest([I2CBase.WRITE, address, 0x66, speed]);
 49 		}
 50 		self.sendI2CRequest([I2CBase.WRITE, address, 0x48, color[0], color[1], color[2]]);
 51 	}
 52 
 53 	this.setFadeSpeed = function(speed) {
 54 		self.sendI2CRequest([I2CBase.WRITE, address, 0x66, speed]);
 55 	}
 56 
 57 	this.playLightScript = function(scriptId, theNumberOfRepeats, lineNumber) {
 58 		theNumberOfRepeats = theNumberOfRepeats || 1;
 59 		self.sendI2CRequest([I2CBase.WRITE, address, 0x70, scriptId, theNumberOfRepeats, lineNumber]);
 60 	}
 61 
 62 	this.stopScript = function() {
 63 		//self.sendI2CRequest([I2CBase.WRITE, address, 'o'.charCodeAt(0)]);
 64 		self.sendI2CRequest([I2CBase.WRITE, address, 0x6F]);
 65 	}
 66 
 67 	// make sure this works
 68 	//public override function handleSysex(command:uint, data:Array):void {
 69 	this.handleI2C = function(command, data) {
 70 		// TODO: implement if needed
 71 		console.log("BlinkM: " + data);
 72 	}
 73 }
 74 
 75 BlinkM.prototype = new I2CBase;
 76 BlinkM.prototype.constructor = BlinkM;
