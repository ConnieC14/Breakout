131 	/**
132 	 * @private
133 	 */
134 	function processData(inputData) {
135 		inputData *= 1;	// force inputData to integer (is there a better way to do this?)
136 		var command;
137 		
138 		// we have command data
139 		if (_waitForData > 0 && inputData < 128) {
140 			_waitForData--;
141 			// collect the data
142 			_storedInputData[_waitForData]=inputData;
143 			// we have all data executeMultiByteCommand
144 			if (_waitForData == 0) {
145 				switch (_executeMultiByteCommand) {
146 					case DIGITAL_MESSAGE:
147 						processDigitalPortBytes(_multiByteChannel, _storedInputData[1], _storedInputData[0]); //(LSB, MSB)
148 						break;
149 					case REPORT_VERSION:
150 						_firmwareVersion=_storedInputData[1] + _storedInputData[0] / 10;
151 						self.dispatchEvent(new ArduinoEvent(ArduinoEvent.FIRMWARE_VERSION, {version: _firmwareVersion}));
152 						break;
153 					case ANALOG_MESSAGE:
154 						_analogData[_multiByteChannel] = self.getValueFromTwo7bitBytes(_storedInputData[1], _storedInputData[0]);
155 						if (_analogData[_multiByteChannel] != _previousAnalogData[_multiByteChannel]) {
156 							self.dispatchEvent(new ArduinoEvent(ArduinoEvent.ANALOG_DATA, {pin: _multiByteChannel, value: _analogData[_multiByteChannel]}));		
157 						}
158 						_previousAnalogData[_multiByteChannel]=_analogData[_multiByteChannel];
159 						break;
160 				}
161 			}
162 		}
163 		// we have SysEx command data
164 		else if (_waitForData < 0) {
165 			// we have all sysex data
166 			if (inputData == END_SYSEX) {
167 				_waitForData = 0;
168 				switch (_sysExData[0]) {
169 					case REPORT_FIRMWARE:
170 						processQueryFirmwareResult(_sysExData);
171 						break;
172 					case STRING_DATA:
173 						processSysExString(_sysExData);
174 						break;
175 					default:
176 						// custom sysEx message
177 						self.dispatchEvent(new ArduinoEvent(ArduinoEvent.SYSEX_MESSAGE, {message: _sysExData}));
178 						break;
179 				}
180 				_sysExData = [];
181 			}
182 			else {
183 				_sysExData.push(inputData);
184 			}
185 		}
186 		// we have a command
187 		else {
188 			// extract the command and channel info from a byte if it is less than 0xF0
189 			if (inputData < 240) {
190 				command = inputData & 240;
191 				_multiByteChannel = inputData & 15;
192 			}
193 			else {
194 				// commands in the 0xF* range don't use channel data
195 				command = inputData;
196 			}
197 			switch (command) {
198 				case REPORT_VERSION:
199 				case DIGITAL_MESSAGE:
200 				case ANALOG_MESSAGE:
201 					_waitForData = 2;	// 2 bytes needed
202 					_executeMultiByteCommand = command;
203 					break;
204 				case START_SYSEX:
205 					_waitForData = -1; // n bytes needed
206 					_executeMultiByteCommand = command;
207 					break;
208 				default:
209 					break;
210 			}
211 		}
212 	}
213 	/**
214 	 * @private
215 	 */
216 	function processDigitalPortBytes(port, bits0_6, bits7_13) {
217 		var low;
218 		var high;
219 		var offset;
220 		// if port is 0, write bits 2..7 into _digitalData[2..7]
221 		// if port is 1, write bits 0..5 into _digitalData[8..13]
222 		if (port == 0) {
223 			low=2;
224 			high=7;
225 			offset=0;
226 		}
227 		else {
228 			low=0;
229 			high=5;
230 			offset=8;
231 		}
232 		var twoBytesForPorts = bits0_6 + (bits7_13 << 7);
233 		var mask;
234 		for (var i = low; i <= high; i++) {
235 			mask=1 << i;
236 			_digitalData[i + offset]=(twoBytesForPorts & mask) >> i;
237 			if (_digitalData[i + offset] != _previousDigitalData[i + offset]) {
238 				self.dispatchEvent(new ArduinoEvent(ArduinoEvent.DIGITAL_DATA, {pin: (i+offset), value: _digitalData[i + offset]}));
239 			}
240 			_previousDigitalData[i + offset]=_digitalData[i + offset];
241 		}	
242 	}
243 	/**
244 	 * @private
245 	 */
246 	function processQueryFirmwareResult(msg) {
247 		var fname ="";
248 		var data;
249 		for (var i = 3; i < msg.length; i+=2)
250 		{
251 			data = String.fromCharCode(msg[i]);
252 			data += String.fromCharCode(msg[i+1]);
253 			fname+=data;
254 		}
255 		_firmwareVersion=msg[1] + msg[2] / 10;
256 		self.dispatchEvent(new ArduinoEvent(ArduinoEvent.FIRMWARE_NAME, {name: fname, version: _firmwareVersion}));
257 		
258 	}
259 	/**
260 	 * @private
261 	 */
262 	function processSysExString(msg) {
263 		var str = "";
264 		var data;
265 		for (var i = 1; i < msg.length; i+=2) {
266 			data = String.fromCharCode(msg[i]);
267 			data += String.fromCharCode(msg[i+1]);
268 			str+=data;
269 		}
270 		self.dispatchEvent(new ArduinoEvent(ArduinoEvent.STRING_MESSAGE, {message: str}));
271 	}

284 	/**
285 	 * A utility class to assemble a single value from the 2 bytes returned from the
286 	 * Arduino (since data is passed in 7 bit Bytes rather than 8 bit it must be 
287 	 * reassembled. This is to be used as a protected method and should not be needed
288 	 * in any application level code.
289 	 *
290 	 * @private
291 	 * @param {Number} lsb The least-significant byte of the 2 values to be concatentated
292 	 * @param {Number} msb The most-significant byte of the 2 values to be concatenated
293 	 * @return {Number} The result of merging the 2 bytes
294 	 */
295 	this.getValueFromTwo7bitBytes = function(lsb, msb) {
296 		return (msb <<7) | lsb;
297 	}

422 	 * Set a digital pin on the Arduino to High or Low.
423 	 *
424 	 * @param {Number} pin The pin number to set or clear.
425 	 * @param {Number} mode Either Arduino.HIGH or Arduino.LOW
426 	 */
427 	this.sendDigital = function(pin, mode) {
428 		if (mode == Arduino.HIGH) {
429 			// set the bit
430 			_digitalPins|=(mode << pin);
431 		}
432 		else if (mode == Arduino.LOW) {
433 			// clear the bit
434 			_digitalPins&=~(1 << pin);
435 		}
436 		if (pin <= 7) {
437 			self.send([DIGITAL_MESSAGE|0, _digitalPins % 128, (_digitalPins >> 7) & 1]);	
438 		}
439 		else {
440 			self.send([DIGITAL_MESSAGE|1, _digitalPins >> 8, 0]);
441 		}
442 	}

467 	/**
468 	 * Send a sysEx message to the Arduino. This is useful for sending custom sysEx data
469 	 * to the Arduino, for example if you are not using StandardFirmata. You would likely
470 	 * use it in a class rather than calling it from your main application.
471 	 *
472 	 * @private
473 	 * @param {Number} command The sysEx command value (see firmata.org)
474 	 * @param {Number[]} data A packet of data representing the sysEx message to be sent
475 	 * @see <a href="http://firmata.org/wiki/Protocol#Sysex_Message_Format">Firmata Sysex Message Format"</a>
476 	 */
477 	this.sendSysex = function(command, data) {
478 		var tempArray = [];
479 		tempArray[0] = START_SYSEX;
480 		tempArray[1] = command;
481 		var charVal;
482 		for (var i=0, len=data.length; i<len; i++) {
483 			charVal = toDec(data[i]);
484 			tempArray.push(charVal % 128);
485 			tempArray.push(charVal >> 7);
486 		}
487 		tempArray.push(END_SYSEX);
488 		self.send(tempArray);		
489 	}

521 		tempArray[0] = START_SYSEX;
522 		tempArray[1] = SERVO_CONFIG;
523 		tempArray[2] = pin;
524 		tempArray[3] = minPulse % 128;
525 		tempArray[4] = minPulse >> 7;
526 		tempArray[5] = maxPulse % 128;
527 		tempArray[6] = maxPulse >> 7;
528 		tempArray[7] = END_SYSEX;
