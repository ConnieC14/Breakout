  5 /**
  6  * Creates a new Servo
  7  *
  8  * @constructor
  9  * @param {Arduino} board A reference to the Arduino class instance.
 10  * @param {Number} servoPin The number of the pin the servo is connected to.
 11  */
 12 function Servo(board, servoPin) {
 13 	"use strict";
 14 	
 15 	var _pin = board.getDigitalPin(servoPin);
 16 	var _angle;
 17 	var self = this;
 18 	
 19 	this.className = "Servo"; 	// for testing
 20 	
 21 	board.sendServoAttach(servoPin);
 22 	
 23 	this.setAngle = function(value) {
 24 		if (_pin.type == Pin.SERVO) {
 25 			_angle = value;
 26 			_pin.setValue(_angle);
 27 		}
 28 	}
 29 	
 30 	this.getAngle = function() {
 31 		if (_pin.type == Pin.SERVO) {
 32 			return _angle;
 33 		}
 34 	}
 35 	
 36 }
