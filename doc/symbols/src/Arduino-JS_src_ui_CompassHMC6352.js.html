1 /**
  2  * HMC6352 digital compass module
  3  *
  4  * @constructor
  5  * @augments I2CBase
  6  * @param {Arduino} board The Arduino instance
  7  * @param {Number} address The i2c address of the compass module
  8  */
  9 function CompassHMC6352(board, address) {
 10 
 11 	address = address || 0x21;
 12 	
 13 	var self = this;
 14 	var _heading = 0;
 15 	var _lastHeading = 0;
 16 	
 17 	I2CBase.call(self, board, address);
 18 		
 19 	// 0x51 = 10 Hz measurement rate, Query mode
 20 	self.sendI2CRequest([I2CBase.WRITE, address, 0x47, 0x74, 0x51]);
 21 	self.sendI2CRequest([I2CBase.WRITE, address, 0x41]);
 22 	
 23 	/**
 24 	 * @returns {Number} The heading in degrees
 25 	 */
 26 	this.getHeading = function() {
 27 		return _heading;
 28 	}
 29 	
 30 	/**
 31 	 * @private
 32 	 */
 33 	this.handleI2C = function(data) {
 34 		// data[0] = register
 35 		_heading = Math.floor(((data[1] << 8) | data[2]) / 10.0);
 36 		
 37 		if (_heading != _lastHeading) {
 38 			self.dispatchEvent(new Event(Event.CHANGE));
 39 		}
 40 		_lastHeading = _heading;
 41 	}
 42 	
 43 	/**
 44 	 * Start continuous reading of the sensor
 45 	 */
 46 	this.startReading = function() {
 47 		self.sendI2CRequest([I2CBase.READ_CONTINUOUS, address, 0x7F, 0x02]);
 48 	}
 49 	
 50 	/**
 51 	 * Stop continuous reading of the sensor
 52 	 */
 53 	this.stopReading = function() {
 54 		self.sendI2CRequest([I2CBase.STOP_READING, address]);
 55 	}
 56 	
 57 	this.startReading();
 58 
 59 }
 60 
 61 CompassHMC6352.prototype = new I2CBase;
 62 CompassHMC6352.prototype.constructor = CompassHMC6352;
