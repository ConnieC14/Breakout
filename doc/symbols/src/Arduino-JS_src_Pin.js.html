1 /**
  2  * An object to represent an Arduino pin
  3  * @constructor
  4  * @param {Number} number The pin number
  5  * @param {Number} type The type of pin
  6  */
  7 function Pin(number, type) {
  8 	this.type = type;
  9 	this.capabilities;
 10 	
 11 	var self = this,
 12 		_number = number,
 13 		_value = 0,
 14 		_lastValue = 0,
 15 		_preFilterValue,
 16 		_average = 0,
 17 		_minimum = Math.pow(2, 16),
 18 		_maximum = 0,
 19 		_avg = 0,
 20 		_sum = 0;
 21 		_numSamples = 0,
 22 		_analogReporting = Pin.OFF,
 23 		MAX_SAMPLES = Number.MAX_VALUE;
 24 	
 25 	var _evtDispatcher = new EventDispatcher(this);
 26 	
 27 	/**
 28 	 * Dispatch a Change event whenever a pin value changes
 29 	 * @private
 30 	 */
 31 	function detectChange(oldValue, newValue) {
 32 		if (oldValue == newValue) return;
 33 		//console.log("detect change");
 34 		self.dispatchEvent(new Event(Event.CHANGE));
 35 	}
 36 	
 37 	/**
 38 	 * From funnel Pin.as
 39 	 * @private
 40 	 */
 41 	function clearWeight() {
 42 		_sum = _average;
 43 		_numSamples = 1;
 44 	}
 45 	
 46 	/**
 47 	 * From funnel Pin.as
 48 	 * @private
 49 	 */
 50 	function calculateMinMaxAndMean(val) {
 51 		_minimum = Math.min(val, _minimum);
 52 		_maximum = Math.max(val, _maximum);
 53 		
 54 		_sum += val;
 55 		_average = _sum / (++_numSamples);
 56 		if (_numSamples >= MAX_SAMPLES) {
 57 			clearWeight();
 58 		}
 59 	}
 60 	
 61 	/**
 62 	 * Get the pin number corresponding to the Arduino documentation for the type of board.
 63 	 * 
 64 	 * @return {Number} The pin number
 65 	 */
 66 	this.getNumber = function() {
 67 		// to do: return number based on pin type?
 68 		return _number;
 69 	}
 70 	
 71 	/**
 72 	 * @return {Number} The average value of the pin over time. Call clear() to reset.
 73 	 */
 74 	this.getAverage = function() {
 75 		return _average;
 76 	}
 77 
 78 	/**
 79 	 * @return {Number} The minimum value of the pin over time. Call clear() to reset.
 80 	 */	
 81 	this.getMinimum = function() {
 82 		return _minimum;
 83 	}
 84 	
 85 	/**
 86 	 * @return {Number} The maximum value of the pin over time. Call clear() to reset.
 87 	 */	
 88 	this.getMaximum = function() {
 89 		return _maximum;
 90 	}
 91 	
 92 	/**
 93 	 * @return {Number} Get the current digital or analog value of the pin
 94 	 */
 95 	this.getValue = function() {
 96 		return _value;
 97 	}
 98 	
 99 	/**
100 	 * @param {Number} val Set the pin value
101 	 */
102 	this.setValue = function(val) {
103 		calculateMinMaxAndMean(val);
104 		_lastValue = _value;
105 		_preFilterValue = val;
106 		_value = val;
107 		detectChange(_lastValue, _value);
108 	}
109 	
110 	this.setAnalogReporting = function(mode) {
111 		_analogReporting = mode;
112 	}
113 	
114 	this.getAnalogReporting = function() {
115 		return _analogReporting;
116 	}
117 	
118 	/**
119 	 * @return {Number} Get the last pin value
120 	 */
121 	this.getLastValue = function() {
122 		return _lastValue;
123 	}
124 	
125 	/**
126 	 * @return {Number} The value before any filters were applied.
127 	 */
128 	this.getPreFilterValue = function() {
129 		return _preFilterValue;
130 	}
131 	
132 	/**
133 	 * Resets the minimum, maximum, average and lastValue of the pin.
134 	 */
135 	this.clear = function() {
136 		_minimum = _maximum = _average = _lastValue = _preFilterValue;
137 		clearWeight();
138 	}
139 	
140 	/* implement EventDispatcher */
141 	
142 	/**
143 	 * @param {String} type The event type
144 	 * @param {Function} listener The function to be called when the event is fired
145 	 */
146 	this.addEventListener = function(type, listener) {
147 		_evtDispatcher.addEventListener(type, listener);
148 	}
149 	
150 	/**
151 	 * @param {String} type The event type
152 	 * @param {Function} listener The function to be called when the event is fired
153 	 */
154 	this.removeEventListener = function(type, listener) {
155 		_evtDispatcher.removeEventListener(type, listener);
156 	}
157 	
158 	/**
159 	 * @param {String} type The event type
160 	 * return {boolean} True is listener exists for this type, false if not.
161 	 */
162 	this.hasEventListener = function(type) {
163 		return _evtDispatcher.hasEventListener(type);
164 	}
165 	
166 	/**
167 	 * @param {Event} type The Event object
168 	 * return {boolean} True if dispatch is successful, false if not.
169 	 */		
170 	this.dispatchEvent = function(event) {
171 		return _evtDispatcher.dispatchEvent(event);
172 	}	
173 }
