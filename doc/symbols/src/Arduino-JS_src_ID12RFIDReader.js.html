  5 /**
  6  * Creates a new ID12RFIDReader
  7  *
  8  * @constructor
  9  */
 10 function ID12RFIDReader(board) {
 11 	"use strict";
 12 
 13 	// to do: object id to be passed in as a param rather than
 14 	// explicitly set? This would support the ability to use multiple
 15 	// RFID readers with a single Arduino.
 16 	// Would also need to update the Arduino library to enable this.
 17 	var 	ID12_READER		= 13,
 18 			READ_EVENT		= 1,
 19 			REMOVE_EVENT	= 2;
 20 
 21 	var self = this;	
 22 	this.className = "ID12RFIDReader"; 	// for testing
 23 	
 24 	var _evtDispatcher = new EventDispatcher(this);
 25 
 26 	board.addEventListener(ArduinoEvent.SYSEX_MESSAGE, onSysExMessage);
 27 	
 28 	// private methods:
 29 	/**
 30 	 * @private
 31 	 */
 32 	function onSysExMessage(event) {
 33 		var message = event.data.message;
 34 
 35 		if (message[0] != ID12_READER) {
 36 			return;
 37 		} else {
 38 			processRFIDData(message);
 39 		}
 40 	}
 41 
 42 	// this is nice! found it here:
 43 	// http://stackoverflow.com/questions/57803/how-to-convert-decimal-to-hex-in-javascript
 44 	/**
 45 	 * @private
 46 	 */
 47 	function dec2hex(i) {
 48    		return (i+0x100).toString(16).substr(-2).toUpperCase();
 49 	}
 50 	
 51 	/**
 52 	 * @private
 53 	 */
 54 	function processRFIDData(data) {
 55 		var tagEvent = board.getValueFromTwo7bitBytes(data[1], data[2]);
 56 		var tagEventType = "";
 57 		var tag = "";
 58 				
 59 		for (var i=3, len=data.length; i<len; i+=2) {
 60 			tag += dec2hex(board.getValueFromTwo7bitBytes(data[i], data[i+1]));
 61 		}
 62 		
 63 		// change this to dispatch a single event and handle add or remove in object parameter?
 64 		if (tagEvent == READ_EVENT) {
 65 			dispatch(new RFIDEvent(RFIDEvent.ADD_TAG, tag));
 66 		} else if (tagEvent == REMOVE_EVENT) {
 67 			dispatch(new RFIDEvent(RFIDEvent.REMOVE_TAG, tag));
 68 		} else {
 69 			// got something else
 70 			return;
 71 		}
 72 		
 73 	}
 74 	
 75 	/**
 76 	 * @private
 77 	 */
 78 	function dispatch(event) {
 79 		self.dispatchEvent(event);
 80 	}
 81 	
 82 	// public methods:
 83 	
 84 	/* implement EventDispatcher */
 85 	
 86 	/**
 87 	 * @param type {String} The event type
 88 	 * @param listener {function} The function to be called when the event is fired
 89 	 */
 90 	this.addEventListener = function(type, listener) {
 91 		_evtDispatcher.addEventListener(type, listener);
 92 	}
 93 	
 94 	/**
 95 	 * @param type {String} The event type
 96 	 * @param listener {function} The function to be called when the event is fired
 97 	 */
 98 	this.removeEventListener = function(type, listener) {
 99 		_evtDispatcher.removeEventListener(type, listener);
100 	}
101 	
102 	/**
103 	 * @param type {String} The event type
104 	 * return {boolean} True is listener exists for this type, false if not.
105 	 */
106 	this.hasEventListener = function(type) {
107 		return _evtDispatcher.hasEventListener(type);
108 	}
109 	
110 	/**
111 	 * @param type {Event} The Event object
112 	 * return {boolean} True if dispatch is successful, false if not.
113 	 */	
114 	this.dispatchEvent = function(event) {
115 		return _evtDispatcher.dispatchEvent(event);
116 	}
117 }
118 
119 /**
120  * @constructor
121  * @augments Event
122  * @param type {string} The event type
123  * @param tag {string} The RFID tag value (hexadecimal)
124  */
125 function RFIDEvent(type, tag) {
126 	this.tag = tag;
127 	// call the super class
128 	// 2nd parameter is passed to EventDispatcher constructor
129 	Event.call(this, type);
130 }
131 
132 RFIDEvent.ADD_TAG = "addTag";
133 RFIDEvent.REMOVE_TAG = "removeTag";
134 
135 // to do: figure out how to inherit a class without using 'new' when we want
136 // to call the super class in the subclass constructor (as we do in this case)
137 RFIDEvent.prototype = new Event;
138 RFIDEvent.prototype.constructor = RFIDEvent;
